## 🧪 **Testing Patterns**

### **User Interaction Testing**

```typescript
import { userEvent } from '@storybook/test';
import { within, waitFor } from '@storybook/test';

export const InteractiveTest: Story = {
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement);

    // Click interactions
    await userEvent.click(canvas.getByLabelText('Content type'));
    await userEvent.click(canvas.getByText('Quick starts'));

    // Keyboard interactions
    await userEvent.keyboard('{Tab}');
    await userEvent.keyboard('{Enter}');

    // Form interactions
    await userEvent.type(canvas.getByLabelText('Search'), 'insights');

    // Wait for async updates
    await waitFor(() => {
      expect(canvas.getByText('Search results')).toBeInTheDocument();
    });

    // Assert final state
    expect(canvas.getByText('5 results found')).toBeInTheDocument();
  },
};
```

### **Accessibility Testing**

```typescript
export const AccessibilityTest: Story = {
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement);

    // Test keyboard navigation
    await userEvent.keyboard('{Tab}');
    expect(canvas.getByLabelText('Content type')).toHaveFocus();

    // Test ARIA attributes
    expect(canvas.getByRole('main')).toHaveAttribute('aria-label', 'Help panel content');

    // Test screen reader announcements
    const liveRegion = canvas.getByRole('status');
    await userEvent.click(canvas.getByText('Quick starts'));

    await waitFor(() => {
      expect(liveRegion).toHaveTextContent(/filtered to show.*quick starts/i);
    });
  },
};
```

### **Error State Testing**

```typescript
export const ErrorState: Story = {
  parameters: {
    msw: {
      handlers: [
        http.get('/api/quickstarts/v1/quickstarts', () => {
          return new HttpResponse(null, { status: 500 });
        }),
      ],
    },
  },

    const canvas = within(canvasElement);

    // Verify error message is displayed
    await waitFor(() => {
      expect(canvas.getByText(/failed to load/i)).toBeInTheDocument();
    });

    // Test retry functionality if available
    const retryButton = canvas.queryByText('Retry');
    if (retryButton) {
      await userEvent.click(retryButton);
    }
  },
};
```

## 📱 **Responsive Testing**

export const MobileView: Story = {
  parameters: {
    viewport: {
      defaultViewport: 'mobile1',

      },

    // Test mobile-specific interactions
    // Touch targets should be at least 44px
    const buttons = canvas.getAllByRole('button');
    buttons.forEach(button => {
      const rect = button.getBoundingClientRect();
      expect(Math.max(rect.width, rect.height)).toBeGreaterThanOrEqual(44);
    });
  },
};

export const OptimizedComponent = ({ resources, filters }) => {

      filters.every(filter => filter.test(resource))
